Tokenization + Normalization

def word_tokenization_normalization(text):
    '''
    This function tokenizes, normalizes, and lemmatizes text using spaCy's NLP model.
    
    This function processes the input text by converting it to lowercase, tokenizing it
    using the spaCy NLP model, and filtering out stopwords, punctuation, numbers, and short words.
    It then lemmatizes each remaining word and returns a list of normalized tokens.

    Parameter
      text (str) - The input text to be tokenized and normalized.

    Return
      list - This returns a list of normalized, lemmatized words (str) that have more
      than two characters and excludes stopwords, punctuation and numbers.
    '''

    text = text.lower() # lowercase
    doc = nlp(text)     # loading text into model

    words_normalized = []
    for word in doc:
        if word.text != '\n' \
        and not word.is_stop \
        and not word.is_punct \
        and not word.like_num \
        and len(word.text.strip()) > 2:
            word_lemmatized = str(word.lemma_)
            words_normalized.append(word_lemmatized)

    return words_normalized

def word_count(word_list):
    '''
    This function creates a dictionary of word counts for a list of words.
    
    This function iterates through the provided word list, converts each word to lowercase,
    and maintains a count of how many times each word appears. This function is typically
    used with output from word_tokenization_normalization.

    Parameter
      word_list(list) - This is a list of words (str) to count.
    
    Return
      dict - A dictionary where keys are words (str) and values are their frequent counts (int)
    '''
    word_counts = {}
    for word in word_list:
      word = word.lower()
      if word in word_counts:
        word_counts[word] += 1
      else:
        word_counts[word] = 1
    return word_counts

def adj_count(text):
  """
  This function extracts and counts adjectives from the input text.
  
  This function uses spaCy's POS tagging to identify adjectives in the text,
  lemmatizes them, and counts their frequency.

  Parameter
    text (str) - The input text to extract adjectives from.
  
  Return
    dict - A dictionary where keys are adjectives (str) and values are their counts (int)
  """
  adj_counts = {}
  
  text = text.lower()
  doc = nlp(text)
  
  for word in doc:
    if word.pos_ == "ADJ" and len(word.text.strip()) > 2:
      adj_lemma = str(word.lemma_).lower()
      if adj_lemma in adj_counts:
        adj_counts[adj_lemma] += 1
      else:
        adj_counts[adj_lemma] = 1
  
  return adj_counts
